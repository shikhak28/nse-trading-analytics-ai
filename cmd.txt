=====================================================================
Windows / PowerShell - streaming setup & verification (new machine)
=====================================================================

# 1. Check Redis (Memurai) is running
Get-Service Memurai
# Status should be "Running"

# 2. Start the API server (in backend/ folder)
node server.js
# Look for: [realtime] socket.io market relay attached

# 3. Start the live-ticker worker (in backend/ folder, separate terminal)
node depthWorker.js
# or: npm run worker:depth
# Look for, in order:
#   [live-ticker] listening for subscription requests
#   [live-ticker] connected to Zerodha
#   [live-ticker] permanently subscribed to <N> tracked companies for depth capture
# If "connected to Zerodha" never shows: finish the Zerodha login in the
# browser first, then wait ~30s for the worker to pick up the new token.

# 4. Verify ticks are flowing (market hours only: 9:15-15:30 IST)
& "C:\Program Files\Memurai\memurai-cli.exe" GET quote:RELIANCE
# Run twice a few seconds apart - "updated_at" should advance, "ltp" should change.
# Swap RELIANCE for any tracked symbol.

# 5. Confirm end-to-end in the browser
# Open the frontend Dashboard / a stock page and check a live tile updates.

# Notes:
# - No ticks outside market hours is expected, not a bug.
# - Zerodha tokens don't refresh; if the connection later drops with auth
#   errors, redo the Zerodha login (Reconnect Zerodha banner).

=====================================================================
Export historical data to CSV
=====================================================================

# Option 1: pgAdmin (GUI)
# Connect to localhost:5432, database stockdb -> Query Tool -> run the
# SELECT above (or any filtered query) -> click the download/export icon
# on the results grid -> CSV.

# Option 2: psql \copy (better for large exports, e.g. a whole symbol's
# 3-year history -- pgAdmin's grid export gets slow/heavy past a few
# hundred thousand rows)
docker exec stock-postgres psql -U admin -d stockdb -c "\copy (SELECT * FROM historical_prices WHERE symbol='RELIANCE' AND interval='day') TO '/tmp/reliance.csv' WITH CSV HEADER"
docker cp stock-postgres:/tmp/reliance.csv ./reliance.csv

# Careful with "all" data: historical_prices has 136M+ rows across all
# 2388 companies and 3 years -- exporting the whole table as one CSV
# would be 10+ GB and impractical to open. Filter by symbol/exchange/date
# range instead (swap the WHERE clause above).
